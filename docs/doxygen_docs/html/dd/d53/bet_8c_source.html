355 int32_t sg777_deckgen_vendor(int32_t playerid, bits256 *cardprods, bits256 *finalcards, int32_t numcards,
356  bits256 *playercards,

358 {

360  static bits256 active_deckid, hash_temp[CARDS777_MAXCARDS];
361  int32_t retval = 1;

364  if (bits256_cmp(deckid, active_deckid) != 0)

372  xoverz = xoverz_donna(curve25519(randcards[i].priv, playercards[i]));
373  vcalc_sha256(0, hash.bytes, xoverz.bytes, sizeof(xoverz));
374  hash_temp[i] = hash; // optimization
375  tmp[i] = fmul_donna(curve25519_fieldelement(hash), randcards[i].priv);
376  }
377 
378  for (int32_t i = 0; i < numcards; i++) {
379  finalcards[i] = tmp[permis_d[i]];
380  g_hash[playerid][i] = hash_temp[permis_d[i]]; // optimization
381  cardprods[i] = randcards[i].prod; // same cardprods[] returned for each player
382  }
383 end:
384  return retval;
385 }
