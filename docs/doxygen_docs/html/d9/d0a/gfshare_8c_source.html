46 void libgfshare_init()
47 {
48  uint32_t i, x = 1;
49  for (i = 0; i < 255; i++) {
50  BET_exps[i] = x;
51  BET_logs[x] = i;
52  x <<= 1;
53  if (x & 0x100)
54  x ^= 0x11d; // Unset the 8th bit and mix in 0x1d
55  }
56  for (i = 255; i < 510; i++)
57  BET_exps[i] = BET_exps[i % 255];
58  BET_logs[0] = 0; // can't log(0) so just set it neatly to 0
59 }

154 // Extract a share from the context. 'share' must be preallocated and at least
155 // 'size' bytes long. 'sharenr' is the index into the 'sharenrs' array of the
156 // share you want.
157 void gfshare_encgetshare(uint8_t *_logs, uint8_t *_exps,
158  struct gfshare_ctx_bet *ctx, uint8_t sharenr,
159  uint8_t *share)
160 {
161  if (_logs == 0)
162  _logs = BET_logs;
163  if (_exps == 0)
164  _exps = BET_exps;
165 
166  uint32_t pos, coefficient, ilog = _logs[ctx->sharenrs[sharenr]];
167  uint8_t *share_ptr, *coefficient_ptr = ctx->buffer;
168 
169  for (pos = 0; pos < ctx->size; pos++)
170  share[pos] = *(coefficient_ptr++);
171  for (coefficient = 1; coefficient < ctx->threshold; coefficient++) {
172  share_ptr = share;
173  for (pos = 0; pos < ctx->size; pos++) {
174  uint8_t share_byte = *share_ptr;
175  if (share_byte != 0)
176  share_byte = _exps[ilog + _logs[share_byte]];
177  *share_ptr++ = share_byte ^ *coefficient_ptr++;
178  }
179  }
180 }

198 // Extract the secret by interpolating the shares. secretbuf must be allocated
199 // and at least 'size' bytes
200 void gfshare_decextract(uint8_t *_logs, uint8_t *_exps,
201  struct gfshare_ctx_bet *ctx, uint8_t *secretbuf)
202 {
203  uint32_t i, j;
204  uint8_t *secret_ptr, *share_ptr, sharei, sharej;
205  if (_logs == 0)
206  _logs = BET_logs;
207  if (_exps == 0)
208  _exps = BET_exps;
209  for (i = 0; i < ctx->size; i++)
210  secretbuf[i] = 0;
211  for (i = 0; i < ctx->sharecount; i++) {
212  // Compute L(i) as per Lagrange Interpolation
213  unsigned Li_top = 0, Li_bottom = 0;
214  if ((sharei = ctx->sharenrs[i]) != 0) {
215  for (j = 0; j < ctx->sharecount; j++) {
216  if (i != j &&
217  sharei != (sharej = ctx->sharenrs[j])) {
218  if (sharej == 0)
219  continue; // skip empty share
220  Li_top += _logs[sharej];
221  if (Li_top >= 0xff)
222  Li_top -= 0xff;
223  Li_bottom += _logs[sharei ^ sharej];
224  if (Li_bottom >= 0xff)
225  Li_bottom -= 0xff;
226  }
227  }
228  if (Li_bottom > Li_top)
229  Li_top += 0xff;
230  Li_top -= Li_bottom; // Li_top is now log(L(i))
231  secret_ptr = secretbuf,
232  share_ptr = ctx->buffer + (ctx->size * i);
233  for (j = 0; j < ctx->size; j++) {
234  if (*share_ptr != 0)
235  *secret_ptr ^= _exps[Li_top +
236  _logs[*share_ptr]];
237  share_ptr++, secret_ptr++;
238  }
239  }
240  }
241 }

243 int32_t gfshare_calc_sharenrs(uint8_t *sharenrs, int32_t N, uint8_t *data,
244  int32_t datasize)
245 {
246  bits256 hash, hash2;
247  uint8_t r;
248  int32_t i, j, n = sizeof(hash);
249  vcalc_sha256(0, hash.bytes, data, datasize);
250  vcalc_sha256(0, hash2.bytes, hash.bytes, sizeof(hash));
251  for (i = 0; i < N; i++) {
252  while (1) {
253  if (n >= sizeof(hash)) {
254  vcalc_sha256(0, hash.bytes, hash2.bytes,
255  sizeof(hash2));
256  vcalc_sha256(0, hash2.bytes, hash.bytes,
257  sizeof(hash));
258  n = 0;
259  }
260  r = hash2.bytes[n++];
261  if ((sharenrs[i] = r) == 0 || sharenrs[i] == 0xff)
262  continue;
263  for (j = 0; j < i; j++)
264  if (sharenrs[j] == sharenrs[i])
265  break;
266  if (j == i)
267  break;
268  }
269  // printf("%3d ",sharenrs[i]);
270  }
271  return (N);
272 }

337 uint8_t *gfshare_recoverdata(uint8_t *shares[], uint8_t *sharenrs, int32_t M,
338  uint8_t *recover, int32_t datasize, int32_t N)
